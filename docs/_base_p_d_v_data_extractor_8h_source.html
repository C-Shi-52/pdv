74typedef struct PDVCONVERTERCORE_API _ModelTreeNodeData
75{
76 NodeObject nodeObject; // 节点对象
77 ModelObject modelObject; // 模型对象, 只有包含网格对象的节点, 该数据有效, 否则为空
78 DftUInt type; // 节点类型, 见ModelTreeNodeType
79 CUnicodeString originID; // 树节点原始ID
80 CUnicodeString originName; // 树节点原始命名
81 CUnicodeString name; // 节点名称
82 PDVMatrix4F localTransform; // 本节点变换到上一层级的局部转换矩阵
83 DftBool visible; // 可见性
84 DftBool external; // 是否关联外部pdv文件
85 CUnicodeString externalPDVPath; // 外部pdv文件路径
86 DftUInt64 cadAnnotationBlockID; // 对应的CAD标注块ID, 如果标注不存在, 该值为DFT_INVALID_ID
87 DftUInt64 pdvAnnotationID; // 对应的pdv标注对象ID, 根据标注块生成标注对象后获得, 如果标注不存在, 该值为DFT_INVALID_ID
88 kernel::pdv::ReferenceGeometryArray referenceGeometryGroup; // 辅助几何集
89 std::vector<_ModelTreeNodeData> subNodes; // 子节点数据
90
91 _ModelTreeNodeData() { Clear(); }
92 _ModelTreeNodeData(const _ModelTreeNodeData& data)
93 {
94 CopyLocalData(data);
95 subNodes = data.subNodes;
96 }
97 _ModelTreeNodeData& operator=(const _ModelTreeNodeData& data)
98 {
99 if (&data != this)
100 {
101 CopyLocalData(data);
102 subNodes = data.subNodes;
103 }
104 return *this;
105 }
106 // 拷贝本层数据
107 void CopyLocalData(const _ModelTreeNodeData& data)
108 {
109 nodeObject = data.nodeObject;
110 modelObject = data.modelObject;
111 type = data.type;
112 originID = data.originID;
113 originName = data.originName;
114 name = data.name;
115 localTransform = data.localTransform;
116 visible = data.visible;
117 external = data.external;
118 externalPDVPath = data.externalPDVPath;
119 cadAnnotationBlockID = data.cadAnnotationBlockID;
120 pdvAnnotationID = data.pdvAnnotationID;
121 referenceGeometryGroup = data.referenceGeometryGroup;
122 subNodes.clear();
123 }
124 // 清除数据
125 void Clear()
126 {
127 nodeObject = NULL;
128 modelObject = NULL;
129 type = NODE_TYPE_NULL;
130 originID = DFT_L_CWCHAR("");
131 originName = DFT_L_CWCHAR("");
132 name = DFT_L_CWCHAR("");
133 localTransform.SetIdentity();
134 visible = FALSE;
135 external = FALSE;
136 externalPDVPath = DFT_L_CWCHAR("");
137 cadAnnotationBlockID = DFT_INVALID_ID;
138 pdvAnnotationID = DFT_INVALID_ID;
139 referenceGeometryGroup.clear();
140 subNodes.clear();
141 }
142 DftUInt GetTotalNodeCount() const;
143 // 输出数据
144 void Output(std::ostream& os, const PDV_CHAR* prefix);
145} ModelTreeNodeData;
