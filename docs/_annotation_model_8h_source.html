27struct PDVCONVERTERCORE_API TextModel
28{
29 CUnicodeString _text; // 文本, 允许多行, 以"\r\n"分隔
30 PDVVector2F _position; // 文本排版的起始位置
31 DftFloat _textHeight; // 文本高度
32 DftFloat _textWidthFactor; // 宽高比
33 DftFloat _angle; // 旋转角度（逆时针, 单位为度）
34 CUnicodeString _fontPath; // 字体文件路径
35
36 TextModel()
37 {
38 _textHeight = 0.f;
39 _textWidthFactor = 0.67f;
40 _angle = 0.f;
41 }
42 TextModel(const TextModel& data)
43 {
44 _text = data._text;
45 _position = data._position;
46 _textHeight = data._textHeight;
47 _textWidthFactor = data._textWidthFactor;
48 _angle = data._angle;
49 _fontPath = data._fontPath;
50 }
51 TextModel& operator=(const TextModel& data)
52 {
53 if (this == &data)
54 return *this;
55 _text = data._text;
56 _position = data._position;
57 _textHeight = data._textHeight;
58 _textWidthFactor = data._textWidthFactor;
59 _angle = data._angle;
60 _fontPath = data._fontPath;
61 return *this;
62 }
63 void Clear()
64 {
65 _text.Empty();
66 _position.SetZero();
67 _textHeight = 0.f;
68 _textWidthFactor = 0.67f;
69 _angle = 0.f;
70 _fontPath.Empty();
71 }
72 // 输出数据
73 void Output(DftJson jsonNode) const;
74};
