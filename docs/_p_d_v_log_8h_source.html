159template <class DataType>
160void JsonOutputData(std::ostream& os, const PDV_CHAR* prefix, const PDV_CHAR* name, const DataType& data, DftBool withComma)
161{
162 CUnicodeString indent0 = prefix;
163 CUnicodeString indent1 = indent0 + JSON_INDENT;
164 os << indent0 << "\"" << name << "\": {" << std::endl;
165 data.Output(os, indent1);
166 os << indent0 << "}" << (withComma ? "," : "") << std::endl;
167}
168
169template<class DataType>
170void JsonOutputDataArray(std::ostream& os, const PDV_CHAR* prefix, const PDV_CHAR* name, const std::vector<DataType>& dataArray, DftBool withComma)
171{
172 CUnicodeString indent0 = prefix;
173 CUnicodeString indent1 = indent0 + JSON_INDENT;
174 CUnicodeString indent2 = indent1 + JSON_INDENT;
175 DftUInt32 count = (DftUInt32)dataArray.size();
176 if (count == 0)
177 os << indent0 << "\"" << name << "(size = " << count << ")\": { }" << (withComma ? "," : "") << std::endl;
178 else
179 {
180 os << indent0 << "\"" << name << "(size = " << count << ")\": {" << std::endl;
181 for (DftUInt32 i = 0; i < count; i++)
182 {
183 os << indent1 << "\"" << name << "[" << i << "]\": {" << std::endl;
184 dataArray[i].Output(os, indent2);
185 os << indent1 << "}" << ((i != count - 1) ? "," : "") << std::endl;
186 }
187 os << indent0 << "}" << (withComma ? "," : "") << std::endl;
188 }
189}

242void JsonOutputDataArray(DftJson jsonNode, const PDV_CHAR* name, const std::vector<DataType>& dataArray)
243{
244 CUnicodeString nodeName;
245 DftUInt count = (DftUInt)dataArray.size();
246 nodeName.Format("%s(size = %u)", name, count);
247 DftJson jsonItemRoot = DftJsonAddObjectToObject(jsonNode, nodeName.ToMultiByte());
248 if (jsonItemRoot)
249 {
250 for (DftUInt i = 0; i < count; i++)
251 {
252 nodeName.Format("%s[%u]", name, i);
253 JsonOutputData(jsonItemRoot, nodeName.ToMultiByte(), dataArray[i]);
254 }
255 }
