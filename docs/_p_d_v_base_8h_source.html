310template <typename T> void ASSIGN(std::vector<T*>& oTarget, const std::vector<T*>& iSource)
311{
312 DftUInt i, count = (DftUInt)oTarget.size();
313 for (i = 0; i < count; i++)
314 SAFE_DELETE(oTarget[i]);
315
316 count = (DftUInt)iSource.size();
317 oTarget.resize(count);
318 for (i = 0; i < count; i++)
319 oTarget[i] = new T(*(iSource[i]));
320}

326template <typename T> void CLEAR(std::vector<T*>& ioTarget)
327{
328 DftUInt i, count = (DftUInt)ioTarget.size();
329 for (i = 0; i < count; i++)
330 SAFE_DELETE(ioTarget[i]);
331 ioTarget.clear();
332}

338template <typename K, typename T> void CLEAR(std::map<K, T*>& ioTarget)
339{
340 typename std::map<K, T *>::iterator iter;
341 for (iter = ioTarget.begin(); iter != ioTarget.end();)
342 {
343 T *value = iter->second;
344 iter = ioTarget.erase(iter);
345 SAFE_DELETE(value);
346 }
